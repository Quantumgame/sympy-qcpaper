circuits = [H(1)*CNOT(1,0), H(1)*CNOT(1,0)*X(1), H(1)*CNOT(1,0)*Z(1), H(1)*CNOT(1,0)*Z(1)*X(1)]

\begin{verbatim}

    display(Eq(circuit*psi,qapply(circuit*psi)))
